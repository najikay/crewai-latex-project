% ── Language Setup ────────────────────────────────────────────────────────────
\setmainlanguage{hebrew}
\setotherlanguage{english}

% Font selection with fallback chain:
%   Try Arial (Windows) → David (Hebrew standard) → FreeSerif (Linux) → default
\IfFontExistsTF{Arial}{%
    \newfontfamily\hebrewfont[Script=Hebrew]{Arial}%
}{%
    \IfFontExistsTF{David}{%
        \newfontfamily\hebrewfont[Script=Hebrew]{David}%
    }{%
        \IfFontExistsTF{FreeSerif}{%
            \newfontfamily\hebrewfont[Script=Hebrew]{FreeSerif}%
        }{%
            \newfontfamily\hebrewfont[Script=Hebrew]{DejaVu Sans}%
        }%
    }%
}
\IfFontExistsTF{Times New Roman}{%
    \newfontfamily\englishfont[Script=Latin]{Times New Roman}%
}{%
    \IfFontExistsTF{FreeSerif}{%
        \newfontfamily\englishfont[Script=Latin]{FreeSerif}%
    }{%
        \newfontfamily\englishfont[Script=Latin]{DejaVu Serif}%
    }%
}

% Hebrew monospace font (required by polyglossia when lstlisting captions contain Hebrew)
\IfFontExistsTF{Courier New}{%
    \newfontfamily\hebrewfonttt[Script=Hebrew]{Courier New}%
}{%
    \IfFontExistsTF{DejaVu Sans Mono}{%
        \newfontfamily\hebrewfonttt[Script=Hebrew]{DejaVu Sans Mono}%
    }{%
        \newfontfamily\hebrewfonttt[Script=Hebrew]{FreeMono}%
    }%
}
% Hebrew sans-serif font (prevents polyglossia error if \textsf used in Hebrew)
\IfFontExistsTF{Arial}{%
    \newfontfamily\hebrewfontsf[Script=Hebrew]{Arial}%
}{%
    \IfFontExistsTF{DejaVu Sans}{%
        \newfontfamily\hebrewfontsf[Script=Hebrew]{DejaVu Sans}%
    }{%
        \newfontfamily\hebrewfontsf[Script=Hebrew]{FreeSans}%
    }%
}

% ── Convenience Commands ──────────────────────────────────────────────────────
% Wrap English technical terms inside Hebrew prose
% \texorpdfstring prevents hyperref from crashing when \textenglish{} appears
% in section titles that get converted to PDF bookmark strings.
\newcommand{\en}[1]{\texorpdfstring{\textenglish{#1}}{#1}}
% Figure reference helper
\newcommand{\figref}[1]{\figurename~\ref{#1}}

% ── Listings Configuration (MUST be LTR — code is always left-to-right) ──────
% The bidi package reverses lstlisting content in Hebrew mode.
% Setting every listing to LTR prevents reversed code/pseudocode.
\lstset{
  basicstyle=\small\ttfamily\LTR,
  keywordstyle=\bfseries\color{blue!70!black},
  commentstyle=\itshape\color{green!50!black},
  stringstyle=\color{red!60!black},
  numbers=left,
  numberstyle=\tiny\color{gray},
  numbersep=8pt,
  frame=single,
  breaklines=true,
  breakatwhitespace=false,
  tabsize=4,
  showstringspaces=false,
  columns=fullflexible,
  xleftmargin=2em,
  framexleftmargin=1.5em,
}

% ── Float spacing (prevent overlapping in two-column IEEE layout) ─────────────
\setlength{\textfloatsep}{10pt plus 2pt minus 2pt}
\setlength{\floatsep}{8pt plus 2pt minus 2pt}
\setlength{\intextsep}{8pt plus 2pt minus 2pt}
% Allow more floats per page to prevent float-queue buildup
\renewcommand{\topfraction}{0.85}

\renewcommand{\textfraction}{0.15}
\renewcommand{\floatpagefraction}{0.65}
\setcounter{topnumber}{3}
\setcounter{bottomnumber}{2}
