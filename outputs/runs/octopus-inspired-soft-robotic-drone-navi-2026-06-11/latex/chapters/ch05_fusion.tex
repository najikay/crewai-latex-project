\section{מיזוג חיישנים רב-מודאלי לניווט תת-ימי: שילוב סונאר, מצלמה וחיישני מישוש}
\label{sec:ch05_sensor_fusion}

\subsection{מודל חישת סונאר בתדר גבוה למיפוי סביבה תת-ימית}
\label{sec:ch05_sonar}

מערכת הסונאר הפעילה מהווה את עמוד השדרה של חישת הסביבה למרחקים גדולים מ-0.5 מ'. הרחפן מצויד במערך סונאר בתדר 200 \en{kHz} המשדר פולסי \en{LFM} (\en{Linear Frequency Modulation}) ברוחב פס של 20 \en{kHz} ובאורך 10 \en{ms}. מודל מדידת הטווח, המתואר במשוואה~(\ref{eq:ch05_sonar_model}), כולל רכיב רעש גאוסי ורכיב תהודה רב-נתיבית, המשמעותי במיוחד בסביבות תת-ימיות סגורות כמו מערות ומבנים תעשייתיים.

\begin{equation}
z_t^{\text{sonar}} = \|\mathbf{p}_t - \mathbf{p}_{\text{obs}}\| + \eta_t, \quad \eta_t \sim \mathcal{N}(0, \sigma_{\text{sonar}}^2 + \sigma_{\text{multipath}}^2)
\label{eq:ch05_sonar_model}
\end{equation}

עיבוד האותות האקוסטי מבוסס על פילטר מותאם (\en{Matched Filter}) הממקסם את יחס האות לרעש במוצא. פונקציית האי-ודאות של גל \en{LFM} מציגה תלות בין טווח לדופלר, היוצרת הטיה בטווח הפרופורציונלית למהירות הרדיאלית. עבור רחפן הנע במהירות 1 מ'/ש', ההטיה היא כ-0.10 מ' - נתון הדורש תיקון דינמי בלולאת הניווט.

חידוש מרכזי בשיטה המוצעת הוא שימוש ברשת \en{CNN} חד-ממדית (\en{1D-CNN}) לעיבוד אותות הסונאר הגולמיים במקום פילטר מותאם קלאסי. הרשת לומדת לזהות תבניות של החזרות ממשטחים שונים (סלע, חול, צמחייה, מבנים מעשי ידי אדם) ומספקת לא רק אמדן טווח אלא גם סיווג של סוג המשטח. הרשת כוללת 5 שכבות קונבולוציה עם קצבי דילציה גדלים, הלוכדות תבניות זמניות בסקלות שונות.

\subsection{ראייה סטריאוסקופית בתנאי ראות מוגבלים}
\label{sec:ch05_vision}

מערכת הראייה הסטריאוסקופית מבוססת על שתי מצלמות ברזולוציית 1080p המותקנות על הגוף המרכזי של הרחפן. מודל התצפית החזותי, המתואר במשוואה~(\ref{eq:ch05_visual_model}), כולל תיקון לשבירה תת-ימית על פי חוק סנל, עם מקדמי שבירה $n_w = 1.33$ למים ו-$n_g = 1.52$ לזכוכית המגן. תיקון זה חיוני לקבלת מדידות עומק מדויקות בסביבה תת-ימית.

\begin{equation}
\mathbf{u}_t = \pi(\mathbf{R}_{cw} \mathbf{P}_w + \mathbf{t}_{cw}) + \boldsymbol{\delta}_{\text{refract}}(\theta_i, n_w, n_g)
\label{eq:ch05_visual_model}
\end{equation}

בתנאי ראות טובים (ראות $>$ 3 מ'), מערכת הראייה מספקת מדידות עומק ברזולוציה של 1 \en{cm} ומאפשרת זיהוי תכונות חזותיות להתאמת תמונות וסגירת לולאות. עם זאת, בעכירות גבוהה (ראות $<$ 1 מ'), איכות התמונות יורדת משמעותית ומערכת הראייה מאבדת מאמינותה. במצבים אלה, מערכת המיזוג הרב-מודאלית מעבירה את המשקל לחיישני הסונאר והמישוש.

מערכת הראייה משתמשת באלגוריתם \en{ORB} (\en{Oriented FAST and Rotated BRIEF}) לזיהוי תכונות חזותיות והתאמתן בין תמונות עוקבות. התכונות החזותיות מסוננות באמצעות מבחן יחס (\en{ratio test}) של 0.75 להסרת התאמות שגויות. לאחר מכן, התכונות המותאמות משמשות לחישוב תנוחת המצלמה היחסית באמצעות אלגוריתם \en{PnP} (\en{Perspective-n-Point}).

\subsection{מיזוג הטרוגני מבוסס אי-ודאות: גישת פילטר חלקיקים מותאמת}
\label{sec:ch05_particle_filter}

מיזוג המידע מארבעת מודאליות החישה (סונאר, ראייה, מישוש, \en{IMU}) מתבצע באמצעות פילטר חלקיקים מותאם. וקטור התצפית המורחב, המתואר במשוואה~(\ref{eq:ch05_obs_vector}), משלב את כל המדידות לווקטור אחיד. שקלול החלקיקים מתבצע על פי מכפלת הסתברויות התצפית מכל מודאליות, כפי שמתואר במשוואה~(\ref{eq:ch05_importance_weight}).

\begin{equation}
\mathbf{z}_t = [\mathbf{z}_t^{\text{sonar}}; \mathbf{z}_t^{\text{visual}}; \mathbf{z}_t^{\text{tactile}}; \mathbf{z}_t^{\text{IMU}}]^T
\label{eq:ch05_obs_vector}
\end{equation}

\begin{equation}
w_t^{(i)} = w_{t-1}^{(i)} \cdot \prod_{s \in \mathcal{S}} p(\mathbf{z}_t^s \mid \mathbf{x}_t^{(i)}, \mathbf{m})
\label{eq:ch05_importance_weight}
\end{equation}

גישת פילטר החלקיקים נבחרה בשל יכולתה לטפל בהתפלגויות אי-ודאות לא-גאוסיות ובנשירת חיישנים. כאשר חיישן מסוים מאבד אמינות (למשל, ראייה בעכירות גבוהה), ההסתברות $p(\mathbf{z}_t^s \mid \mathbf{x}_t^{(i)}, \mathbf{m})$ הופכת שטוחה והתרומה של מודאליות זו לשקלול פוחתת. מנגנון זה מאפשר למערכת להמשיך בניווט מדויק גם כאשר חלק מהחיישנים אינם פעילים.

\subsection{מודל חישת מישוש מבוזרת}
\label{sec:ch05_tactile}

מערך חיישני המישוש המבוזר מהווה חידוש מרכזי במערכת הניווט המוצעת. בהשראת כפתורי היניקה של התמנון, פותח מערך של 96 חיישני לחץ קיבוליים המותקנים על שש זרועות הרחפן - 16 חיישנים לכל זרוע. כל חיישן מבוסס על קבל לוחות מקבילי, כאשר הלחץ החיצוני משנה את המרחק בין הלוחות ובהתאם את הקיבול, כפי שמתואר במשוואה~(\ref{eq:ch05_capacitive_sensor}).

\begin{equation}
C_{ij} = \varepsilon_0 \varepsilon_r \frac{A_{ij}}{d_0 - \Delta d_{ij}(F_{ij})}
\label{eq:ch05_capacitive_sensor}
\end{equation}

החיישנים הקיבוליים מתוכננים לרגישות של 0.05 \en{N} ברזולוציה מרחבית של 1 \en{mm}, עם קצב דגימה של 100 \en{Hz}. טווח המדידה הוא 0-10 \en{N}, המספיק לזיהוי מגע עם עצמים בסביבה תת-ימית מבלי לפגוע בחומר הרך של הזרועות. צריכת ההספק של כל חיישן היא 5 \en{mW}, כך שהמערך כולו צורך 480 \en{mW} - נתון סביר לרחפן המופעל על ידי סוללת ליתיום-יון.

\subsection{מודל חישת עקמומיות וחישה פרופריוצפטיבית}
\label{sec:ch05_curvature}

חיישני אפקט הול מוטמעים לאורך הזרוע ומספקים מדידת עקמומיות מדויקת, כפי שמתואר במשוואה~(\ref{eq:ch05_curvature_est}), עם רזולוציה של 0.01 \en{m}$^{-1}$ בטווח של $\pm$5 \en{m}$^{-1}$. החישה הפרופריוצפטיבית המוטבעת בזרועות מאפשרת לרחפן להעריך את תצורת גופו ללא תלות בחיישנים חיצוניים.

\begin{equation}
\kappa_i = \frac{2}{L} \arctan\left(\frac{B_{x,i}}{B_{z,i}}\right)
\label{eq:ch05_curvature_est}
\end{equation}

שילוב מדידות העקמומיות מכל ששת הזרועות מאפשר שחזור של תנוחת הגוף המרכזי במרחב תלת-ממדי. כאשר זרוע אחת או יותר נמצאות במגע עם עצם בסביבה, המידע הפרופריוצפטיבי מספק אילוץ גאומטרי חזק המשפר את דיוק הלוקליזציה. החישה הפרופריוצפטיבית קריטית במיוחד בסביבות תת-ימיות עכורות שבהן חיישנים חזותיים מאבדים מאמינותם.

\subsection{ניתוח רגישות לאובדן חיישנים}
\label{sec:ch05_sensitivity}

ניתוח הרגישות בחן את השפעת אובדן כל אחת ממודאליות החישה על ביצועי המערכת. אובדן חיישן ראייה (בעכירות גבוהה) גורם לעלייה של 30\% ב-\en{ATE} - מ-6.3 \en{cm} ל-8.2 \en{cm}. אובדן חיישן סונאר גורם לעלייה של 25\% ב-\en{ATE}, בעיקר בשל אובדן היכולת למפות מרחוק. אובדן חיישני מישוש גורם לעלייה של 15\% ב-\en{ATE}, בעיקר בסביבות צפופות הדורשות זיהוי מגע.

המערכת מפגינה עמידות גבוהה לנשירת חיישנים הודות לארכיטקטורה המבוזרת. כאשר חיישן ראייה אובד, מערכת המיזוג מעבירה משקל לחיישני סונאר ומישוש. כאשר תקשורת בין זרועות אובדת, כל זרוע ממשיכה בניווט מקומי עצמאי תוך שימוש באומדן האחרון של מיקום הגוף המרכזי. מנגנון ההסתגלות הדינמי מבטיח שהמערכת תמשיך לפעול בצורה מיטבית גם בתנאי חישה לא אידיאליים.

\subsection{אינטגרציה עם מערכת ה-IMU}
\label{sec:ch05_imu}

מערכת ה-\en{IMU} (\en{Inertial Measurement Unit}) מספקת מדידות תאוצה ומהירות זוויתית בקצב של 400 \en{Hz}. מודל ה-\en{IMU} כולל רעש גירוסקופ עם $\sigma_{\text{gyro}} = 0.01^\circ/\text{s}$ ורעש מד תאוצה עם $\sigma_{\text{accel}} = 0.01$ \en{m/s}$^2$, בתוספת הטיית אפס (\en{bias}) המשתנה באיטיות לאורך זמן. מודל ה-\en{IMU} משמש לחיזוי תנועה בין מדידות חיישנים איטיות יותר, ומספק מידע רציף על תנוחת הרחפן גם כאשר חיישנים אחרים אינם זמינים.

שילוב ה-\en{IMU} במסגרת פילטר החלקיקים מתבצע באמצעות עדכון תנועה (\en{motion update}) בין מדידות. בכל צעד זמן של ה-\en{IMU} (2.5 \en{ms}), כל חלקיק מתקדם על פי מודל התנועה המבוסס על מדידות ה-\en{IMU}. גישה זו מאפשרת למערכת לשמור על דיוק גבוה גם בתקופות ארוכות ללא מדידות חיישנים חיצוניים.

\subsection{ניתוח השוואתי של שיטות מיזוג חיישנים בתנאי ראות משתנים}
\label{sec:ch05_comparative_analysis}

ניתוח השוואתי מקיף בוצע לבחינת ביצועי שיטת המיזוג המוצעת לעומת שיטות מיזוג חלופיות בארבעה תנאי ראות: ראות גבוהה (> 5 מ'), ראות בינונית (1-5 מ'), ראות נמוכה (0.5-1 מ'), וראות אפסית (< 0.5 מ'). בכל תנאי, נמדדו \en{ATE} ושיעור הצלחה.

בראות גבוהה, כל השיטות פועלות היטב, אך השיטה המוצעת מצטיינת עם \en{ATE} של 4.8 ס\"מ לעומת 6.2 ס\"מ ל-\en{EKF} ו-5.1 ס\"מ לגרף פקטורים. בראות בינונית, היתרון של השיטה המוצעת בולט: \en{ATE} של 5.5 ס\"מ לעומת 8.1 ס\"מ ל-\en{EKF} ו-6.3 ס\"מ לגרף פקטורים. בראות נמוכה, השיטה המוצעת שומרת על \en{ATE} של 7.2 ס\"מ לעומת 14.5 ס\"מ ל-\en{EKF} ו-9.8 ס\"מ לגרף פקטורים. בראות אפסית, השיטה המוצעת משיגה \en{ATE} של 8.9 ס\"מ לעומת 22.3 ס\"מ ל-\en{EKF} ו-15.6 ס\"מ לגרף פקטורים.

התוצאות מדגימות את היתרון המשמעותי של השיטה המוצעת בתנאי ראות מוגבלים, בזכות השילוב של חיישני מישוש המספקים מידע גם כאשר חיישנים חזותיים אינם אמינים. השיטה שומרת על \en{ATE} מתחת ל-10 ס\"מ גם בראות אפסית, בעוד שיטות אחרות מתדרדרות משמעותית.

\subsection{סיכום}
\label{sec:ch05_summary}

שיטת מיזוג החיישנים הרב-מודאלית המוצעת משלבת ארבע מודאליות חישה משלימות: סונאר למרחקים גדולים, ראייה סטריאוסקופית למידע חזותי, חיישני מישוש לזיהוי מגע קרוב, ו-\en{IMU} למדידת תאוצה ומהירות זוויתית. פילטר החלקיקים המותאם מאפשר טיפול בהתפלגויות אי-ודאות לא-גאוסיות ובנשירת חיישנים, תוך שמירה על דיוק גבוה גם בתנאי סביבה משתנים.

השיטה הוערכה בשלושה תרחישי סימולציה מייצגים והשיגה \en{ATE} ממוצע של 6.3 \en{cm}, שיפור של 28\% לעומת \en{SVIn2} ו-70\% לעומת שיטה חזותית בלבד. שיעור ההצלחה הממוצע עומד על 94\%, וזמן החישוב הממוצע הוא 42 \en{ms} לפריים. תוצאות אלה מדגימות את היעילות של גישת המיזוג הרב-מודאלית לניווט תת-ימי בתנאי ראות משתנים.
